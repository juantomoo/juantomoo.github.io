add_action( 'init', 'machina_register_cpt' );

function machina_register_cpt() {
    register_post_type( 'portfolio', [
        'public' => true,
        'label'  => 'Portfolio',
        'supports' => ['title', 'editor', 'thumbnail']
    ]);
}
